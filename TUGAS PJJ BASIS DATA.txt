
1. CREATE DATABASE db_sekolah
2. CREATE TABLE jurusan(
    kode_jurusan VARCHAR(10) PRIMARY KEY,
    Nama_jurusan VARCHAR(50),
    Keterangan VARCHAR(50) );
    
CREATE TABLE siswa(
    nis VARCHAR(10) PRIMARY KEY,
    Nama_siswa VARCHAR(50),
    Jenis_kelamin VARCHAR(50),
    Tgl_lahir DATE,
    Alamat VARCHAR(50),
    kode_jurusan VARCHAR(10),
    FOREIGN KEY (kode_jurusan) REFERENCES jurusan(kode_jurusan) );
    
CREATE TABLE guru(
    nip VARCHAR(10) PRIMARY KEY,
    Nama_guru VARCHAR(50),
    Mapel_diajar VARCHAR(50) );
    
CREATE TABLE mata_pelajaran(
    kode_mapel VARCHAR(10) PRIMARY KEY,
    Nama_mapel VARCHAR(50),
    Jumlah_jam INT,
    kode_jurusan VARCHAR(10),
    FOREIGN KEY (kode_jurusan) REFERENCES jurusan(kode_jurusan) );
    
CREATE TABLE nilai(
    id_nilai VARCHAR(10) PRIMARY KEY,
    nis VARCHAR(10),
    kode_mapel VARCHAR(10),
    nilai INT,
    FOREIGN KEY (nis) REFERENCES siswa(nis),
    FOREIGN KEY (kode_mapel) REFERENCES mata_pelajaran(kode_mapel) );

A. Menampilkan semua siswa yang berasal dari jurusan “Rekayasa Perangkat Lunak”. 
=SELECT * FROM siswa WHERE kode_jurusan = 'RPL02';
=SELECT * FROM 
siswa AS S
JOIN
jurusan AS J
ON
S.kode_jurusan = J.kode_jurusan
WHERE J.Nama_jurusan = 'Rekayasa Perangkat Lunak';

B. Menampilkan semua mata pelajaran dengan jumlah jam lebih dari 4. 
=SELECT Nama_mapel 
FROM
mata_pelajaran
WHERE Jumlah_jam > 4;

C. Menampilkan semua nilai siswa yang lebih besar dari 80. 
=SELECT Nama_siswa, nilai
FROM
siswa AS S
JOIN 
nilai AS N
ON
S.nis = N.nis
WHERE nilai > 80;

A. Menampilkan daftar siswa beserta nama jurusan mereka (JOIN siswa dengan jurusan).

siswa AS S
JOIN
jurusan AS J
ON
S.kode_jurusan = J.kode_jurusan;

B.Menampilkan daftar nilai siswa (nama siswa, nama mapel, dan nilai).

FROM
siswa AS S
JOIN
nilai AS N
ON
S.nis = N.nis

mata_pelajaran AS M
ON
N.kode_mapel = M.kode_mapel;
