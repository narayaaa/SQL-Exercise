2. SELECT CONCAT ("Mr. ", customerName), customerGender, customerAddress

3.SELECT MP.ProductName, MC.categoryName, CONCAT (" $ ", MP.ProductPrice) AS "ProductPrice"
FROM msproduct AS MP
JOIN
mscategory AS MC
ON 
MC.categoryID = MP.CategoryID
WHERE categoryName = "Dairy";

msproduct

productName LIKE "%Cheese";

5. SELECT DISTINCT employeeName, employeeAddress

msemployee AS ME

transactionheader AS TH

ME.EmployeeID = TH.EmployeeID

MONTH (TH.TransactionDate) = 12;

7. SELECT DISTINCT MC.customerName, ME.employeeName, TH.transactionID, TH.transactionDate
FROM mscustomer AS MC
JOIN
transactionheader AS TH
ON
MC.CustomerID = TH.CustomerID
JOIN
msemployee AS ME
ON 
TH.EmployeeID = ME.EmployeeID
JOIN
transactiondetail AS TD
ON 
TH.TransactionID = TD.TransactionID
JOIN
msproduct AS MP
ON
TD.ProductID = MP.ProductID
JOIN 
mscategory AS MCA
ON
MP.CategoryID = MCA.CategoryID
WHERE
ME.EmployeeGender = "Male" AND MCA.CategoryName  = "Meat"
ORDER BY ( MC.CustomerName) ASC;
